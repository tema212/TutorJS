• Для создания бегущей строки в Html используется команда Marquee


<marquee> Бегущая строка</marquee>


•Для создания бегущей строки в Js скрипт будет подлинее)


!Doctype Html

< a id="ran" class="text">Бегущая строка </a>

!Doctype Js

function animate(id) {

  let node =

  document.getElementById(id).childNodes[0];

let text = node.data;

setInterval(function () {

text = text.substring(1) + text[0];

node.data = text;

}, 175);

}

window.addEventListener('load', function (e) { animate('ran'); }, false);
